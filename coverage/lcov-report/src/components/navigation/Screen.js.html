import React from "react"
import Classnames from 'classnames'
import Boundary from "./util/Boundary"
import ScreenBoundary from "./util/ScreenBoundary"
import "../../stylesheets/navigation/screen.css"
 
const X_AXIS = 'X'
const Y_AXIS = 'Y'
 
export default class Screen extends React.Component {
  constructor(props) {
    super(props)
    this.handleBoundary = this.handleBoundary.bind(this)
    this.incrementLocals = this.incrementLocals.bind(this)
    this.onSectionEntered = this.onSectionEntered.bind(this)
    this.onFocus = this.onFocus.bind(this)
    this.state = {focusX: 0, focusY: 0, sectionX: 0, sectionY: 0, currentSection: null}
  }
 
  onFocus(item) {
    this.props.updateCurrentItem(item)
    this.props.onFocus(item)
  }
 
  onSectionEntered(section) {
    const { currentSection: previousSection } = this.state
    this.setState({currentSection: section})
    return this.props.onSectionEntered(previousSection, section)
  }
 
  pastBoundary(x, y) {
    const { sections } = this.props
    return ((x < 0 || y < 0) || x >= sections.length || y >= sections.sectionAt(x).length)
  }
 
  handleBoundary(vectorX, vectorY) {
    const { sectionX, sectionY } = this.state
    let nextSectionX = sectionX + vectorX
    let nextSectionY = sectionY + vectorY
    let parentVectors
 
    const finishHandleBoundary = ((parentVectors) => {
      let [parentVectorX, parentVectorY] = parentVectors
      nextSectionX = sectionX + parentVectorX
      nextSectionY = sectionY + parentVectorY
 
      this.setState({sectionX: nextSectionX, sectionY: nextSectionY})
    }).bind(this)
 
    if (this.pastBoundary(nextSectionX, nextSectionY)) {
      const promiseOrParentVectors = this.props.handleBoundary(vectorX, vectorY)
 
      if (!!promiseOrParentVectors.then) {
        promiseOrParentVectors.then((resolvedParentVectors) => {
          finishHandleBoundary(resolvedParentVectors)
        })
        return
      }
      parentVectors = promiseOrParentVectors
    } else {
      parentVectors = this.props.onNextSection(vectorX, vectorY, this.state.currentSection)
    }
 
    finishHandleBoundary(parentVectors)
  }
 
  incrementLocals(vectorX, vectorY) {
    const { focusX, focusY } = this.state
    this.setState({focusX: focusX + vectorX, focusY: focusY + vectorY})
  }
 
  render() {
    const { globalX, globalY, hasFocus, sections, classNames, startVideoPlayer } = this.props
    const { sectionX, sectionY, focusX, focusY } = this.state
    const { handleBoundary, incrementLocals, onFocus, onSectionEntered } = this
 
    const handlers = {handleBoundary, incrementLocals, onFocus, onSectionEntered, startVideoPlayer}
    const coordinates = {globalX, globalY, focusX: focusX, focusY: focusY}
    const className = Classnames({
      screen: true,
      active: hasFocus,
      ...classNames
    })
 
    let sectionProps, key, mapper, children
 
    return (
      <div className={className}>
        {sections.map((matrixRow, x) => {
          // TODO: investigate performance of this? this is n^3, which may be a bit expensive if the # of sections is high enough
          return matrixRow.map((boundary, z) => {
            children = boundary.map((section, y) => {
              key = `section-${x}-${y}-${z}`
              sectionProps = {...section.props, hasFocus: true,  ...coordinates, ...handlers}
              if (sections.sectionAt(sectionX, sectionY) === section && hasFocus) {
                return {...section, key, props: sectionProps}
              }
              return {...section, key}
            })
            return boundary.wrappedChildren(children)
          })
        })}
      </div>
    )
  }
}
